131 - (unsigned)addTypes:(CPArray)types owner:(id)anOwner
132 {
133  var i = 0,
134  count = types.length;
135 
136  for (; i < count; ++i)
137  {
138  var type = types[i];
139 
140  if (![_owners objectForKey:type])
