480 var _CPFontConcatNameWithFallback = function(aName)
481 {
482  var names = _CPFontNormalizedNameArray(aName),
483  fallbackFaces = _CPFontFallbackFaces.slice(0);
484 
485  // Remove the fallback names used in the names passed in
486  for (var i = 0; i < names.length; ++i)
487  {
488  for (var j = 0; j < fallbackFaces.length; ++j)
489  {
490  if (names[i].toLowerCase() === fallbackFaces[j].toLowerCase())
491  {
492  fallbackFaces.splice(j, 1);
493  break;
494  }
495  }
496 
497  if (names[i].indexOf(" ") > 0)
498  names[i] = '"' + names[i] + '"';
499  }
500 
501  return names.concat(fallbackFaces).join(", ");
502 };
