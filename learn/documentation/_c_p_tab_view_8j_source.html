283 - (BOOL)selectTabViewItemAtIndex:(CPUInteger)anIndex
284 {
285  if (anIndex === _selectedIndex)
286  return;
287 
288  var aTabViewItem = [self tabViewItemAtIndex:anIndex];
289 
290  if ((_delegateSelectors & CPTabViewShouldSelectTabViewItemSelector) && ![_delegate tabView:self shouldSelectTabViewItem:aTabViewItem])
291  return NO;
292 
293  if (_delegateSelectors & CPTabViewWillSelectTabViewItemSelector)
294  [_delegate tabView:self willSelectTabViewItem:aTabViewItem];
295 
296  [_tabs selectSegmentWithTag:anIndex];
297  [self _setSelectedIndex:anIndex];
298 
299  if (_delegateSelectors & CPTabViewDidSelectTabViewItemSelector)
300  [_delegate tabView:self didSelectTabViewItem:aTabViewItem];
301 
302  return YES;
303 }

560 - (void)encodeWithCoder:(CPCoder)aCoder
561 {
562  // Don't bother to encode the CPBox. We will recreate it on decode and its content view is already
563  // stored by the tab view item. Not encoding _box makes the resulting archive smaller and reduces
564  // the surface for decoding bugs (of which we've had many in tab view).
565  var subviews = [self subviews];
566  [_box removeFromSuperview];
567  [super encodeWithCoder:aCoder];
568  [self setSubviews:subviews];
569 
570  [aCoder encodeObject:_items forKey:CPTabViewItemsKey];
571 
572  var selected = [self selectedTabViewItem];
573  if (selected)
574  [aCoder encodeObject:selected forKey:CPTabViewSelectedItemKey];
575 
576  [aCoder encodeInt:_type forKey:CPTabViewTypeKey];
577  [aCoder encodeObject:_font forKey:CPTabViewFontKey];
578 
579  [aCoder encodeConditionalObject:_delegate forKey:CPTabViewDelegateKey];
580 }
