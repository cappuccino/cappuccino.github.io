597 - (void)_updateSearchMenu
598 {
599  if (_searchMenuTemplate === nil)
600  return;
601 
602  var menu = [[CPMenu alloc] init],
603  countOfRecents = [_recentSearches count],
604  numberOfItems = [_searchMenuTemplate numberOfItems];
605 
606  for (var i = 0; i < numberOfItems; i++)
607  {
608  var item = [[_searchMenuTemplate itemAtIndex:i] copy];
609 
610  switch ([item tag])
611  {
612  case CPSearchFieldRecentsTitleMenuItemTag:
613  if (countOfRecents === 0)
614  continue;
615 
616  if ([menu numberOfItems] > 0)
617  [self _addSeparatorToMenu:menu];
618  break;
619 
620  case CPSearchFieldRecentsMenuItemTag:
621  {
622  var itemAction = @selector(_searchFieldSearch:);
623 
624  for (var recentIndex = 0; recentIndex < countOfRecents; ++recentIndex)
625  {
626  // RECENT_SEARCH_PREFIX is a hack until CPMenuItem -setIndentationLevel works
627  var recentItem = [[CPMenuItem alloc] initWithTitle:RECENT_SEARCH_PREFIX + [_recentSearches objectAtIndex:recentIndex]
628  action:itemAction
629  keyEquivalent:[item keyEquivalent]];
630  [item setTarget:self];
631  [menu addItem:recentItem];
632  }
633 
634  continue;
635  }
636 
637  case CPSearchFieldClearRecentsMenuItemTag:
638  if (countOfRecents === 0)
639  continue;
640 
641  if ([menu numberOfItems] > 0)
642  [self _addSeparatorToMenu:menu];
643 
644  [item setAction:@selector(_searchFieldClearRecents:)];
645  [item setTarget:self];
646  break;
647 
648  case CPSearchFieldNoRecentsMenuItemTag:
649  if (countOfRecents !== 0)
650  continue;
651 
652  if ([menu numberOfItems] > 0)
653  [self _addSeparatorToMenu:menu];
654  break;
655  }
656 
657  [item setEnabled:([item isEnabled] && [item action] != nil && [item target] != nil)];
658  [menu addItem:item];
659  }
660 
661  [menu setDelegate:self];
662 
663  _searchMenu = menu;
664 }
