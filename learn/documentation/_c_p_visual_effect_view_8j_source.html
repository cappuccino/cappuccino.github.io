118  var dark = [[self appearance] isEqual:[CPAppearance appearanceNamed:CPAppearanceNameVibrantDark]],
119  prop = CPBrowserStyleProperty("backdrop-filter"),
120  color = (dark ? [CPColor colorWithHexString:@"1e1e1e"] : [CPColor whiteColor]),

131 - (void)_applyVibrancyState
132 {
133  switch (_state)
134  {
135  case CPVisualEffectStateFollowsWindowActiveState:
136  [self _setEffectEnabled:[self hasThemeState:CPThemeStateKeyWindow]];
137  break;
138 
139  case CPVisualEffectStateActive:
140  [self _setEffectEnabled:YES];
141  break;
142 
143  case CPVisualEffectStateInactive:
144  [self _setEffectEnabled:NO];
145  break;
146  }
147 }
