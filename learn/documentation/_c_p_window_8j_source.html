417  // and can apply the positioning mask, if any, from the nib.
418  if (_positioningScreenRect)
419  {
420  var actualScreenRect = [CPPlatform isBrowser] ? [_platformWindow contentBounds] : [[self screen] visibleFrame],
421  frame = [self frame],

423

750  Constrain a frame so that the window remains at least partially visible on screen,
751  moving or resizing the frame as necessary.
752 */
753 - (CGRect)_constrainFrame:(CGRect)aFrame toUsableScreenWidth:(BOOL)constrainWidth andHeight:(BOOL)constrainHeight
754 {
755  var frame = CGRectMakeCopy(aFrame);
756 
757  if (!_constrainsToUsableScreen || !_isVisible)
758  return frame;
759 
760  var usableRect = [_platformWindow usableContentFrame];
761 
762  if (constrainWidth)
763  {
764  // First move the frame right to ensure the left side is within the usable rect.
765  frame.origin.x = MAX(frame.origin.x, usableRect.origin.x);
766 
767  // Now move the frame left so that the right side is within the usable rect.
768  var maxX = MIN(CGRectGetMaxX(frame), CGRectGetMaxX(usableRect));
769  frame.origin.x = maxX - frame.size.width;
770 
771  // Finally, adjust the left + width to ensure the left side is within the usable rect.
772  var usableWidth = CGRectGetWidth(usableRect);
773 
774  if (CGRectGetWidth(frame) > usableWidth)
775  {
776  frame.origin.x = CGRectGetMinX(usableRect);
777  frame.size.width = MAX(usableWidth, _minSize.width);
778  }
779  }
780 
781  if (constrainHeight)
782  {
783  // First move the frame down to ensure the top is within the usable rect.
784  frame.origin.y = MAX(frame.origin.y, usableRect.origin.y);
785 
786  // Now move the frame up so that the bottom is within the usable rect.
787  var maxY = MIN(CGRectGetMaxY(frame), CGRectGetMaxY(usableRect));
788  frame.origin.y = maxY - frame.size.height;
789 
790  // Finally, adjust the top + height to ensure the top is within the usable rect.
791  var usableHeight = CGRectGetHeight(usableRect);
792 
793  if (CGRectGetHeight(frame) > usableHeight)
794  {
795  frame.origin.y = CGRectGetMinY(usableRect);
796  frame.size.height = MAX(usableHeight, _minSize.height);
797  }
798  }
799 
800  return frame;
801 }

803 /*
804  Constrain the origin of a frame such that:
805 
806  - The window view's minimum resize width is kept onscreen at the left/right of the window.
807  - The top of the window is kept below the top of the usable content.
808  - The top of the contentView + CPWindowMinVisibleVerticalMargin is kept above the bottom of the usable content.
809 */
810 - (CGRect)_constrainOriginOfFrame:(CGRect)aFrame
811 {
812  var frame = CGRectMakeCopy(aFrame);
813 
814  if (!_constrainsToUsableScreen || !_isVisible)
815  return frame;
816 
817  var usableRect = [_platformWindow usableContentFrame],
818  minimumSize = [_windowView _minimumResizeSize];
819 
820  // First constrain x so that at least CPWindowMinVisibleHorizontalMargin is visible on the right
821  frame.origin.x = MAX(frame.origin.x, CGRectGetMinX(usableRect) + minimumSize.width - CGRectGetWidth(frame));
822 
823  // Now constrain x so that at least CPWindowMinVisibleHorizontalMargin is visible on the left
824  frame.origin.x = MIN(frame.origin.x, CGRectGetMaxX(usableRect) - minimumSize.width);
825 
826  // Now constrain y so that it is below the top of the usable content
827  frame.origin.y = MAX(frame.origin.y, CGRectGetMinY(usableRect));
828 
829  // Finally constrain y so that at least CPWindowMinVisibleHorizontalMargin is visible at the bottom
830  frame.origin.y = MIN(frame.origin.y, CGRectGetMaxY(usableRect) - CGRectGetMinY([_contentView frame]) - CPWindowMinVisibleVerticalMargin);
831 
832  return frame;
833 }
