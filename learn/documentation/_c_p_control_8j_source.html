368 - (void)trackMouse:(CPEvent)anEvent
369 {
370  var type = [anEvent type],
371  currentLocation = [self convertPoint:[anEvent locationInWindow] fromView:nil],
372  isWithinFrame = [self tracksMouseOutsideOfFrame] || CGRectContainsPoint([self bounds], currentLocation);
373 
374  if (type === CPLeftMouseUp)
375  {
376  [self stopTracking:_previousTrackingLocation at:currentLocation mouseIsUp:YES];
377 
378  _trackingMouseDownFlags = 0;
379 
380  if (isWithinFrame)
381  [self setThemeState:CPThemeStateHovered];
382  }
383  else
384  {
385  [self unsetThemeState:CPThemeStateHovered];
386 
387  if (type === CPLeftMouseDown)
388  {
389  _trackingMouseDownFlags = [anEvent modifierFlags];
390  _continuousTracking = [self startTrackingAt:currentLocation];
391  }
392  else if (type === CPLeftMouseDragged)
393  {
394  if (isWithinFrame)
395  {
396  if (!_trackingWasWithinFrame)
397  _continuousTracking = [self startTrackingAt:currentLocation];
398 
399  else if (_continuousTracking)
400  _continuousTracking = [self continueTracking:_previousTrackingLocation at:currentLocation];
401  }
402  else
403  [self stopTracking:_previousTrackingLocation at:currentLocation mouseIsUp:NO];
404  }
405 
406  [CPApp setTarget:self selector:@selector(trackMouse:) forNextEventMatchingMask:CPLeftMouseDraggedMask | CPLeftMouseUpMask untilDate:nil inMode:nil dequeue:YES];
407  }
408 
409  if ((_sendActionOn & (1 << type)) && isWithinFrame)
410  [self sendAction:_action to:_target];
411 
412  _trackingWasWithinFrame = isWithinFrame;
413  _previousTrackingLocation = currentLocation;
414 }

630 - (void)setStringValue:(CPString)aString
631 {
632  // Cocoa raises an invalid parameter assertion and returns if you pass nil.

651  }
652  else
